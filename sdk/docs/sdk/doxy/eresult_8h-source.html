00008 //
00009 // Every distinct module should define its own zone for eresults,
00010 // and a set of codes within that zone. The code by itself is not
00011 // unique, in that every zone will likely define error codes that
00012 // start at 0 and increment. the combination of zone+code, however,
00013 // is certainly unique
00014 //
00015 
00016 typedef int ERESULT;
00017 
00018 #define MAKE_ERESULT( severity, zone, code )        \
00019         (ERESULT)(((unsigned int)(severity)<<24) |  \
00020                   ((unsigned int)(zone)<<16)     |  \
00021                   ((unsigned int)(code)))
00022 
00023 #define SUCCEEDED( res ) ((ERESULT)(res) >= 0)
00024 #define FAILED( res )    ((ERESULT)(res) <  0)
00025 
00026 #define SEVERITY( res )  (ERESULT)((unsigned int)(res) >> 24)
00027 #define ZONE( res )      (ERESULT)(((unsigned int)(res) >> 16) & 0xff)
00028 #define CODE( res )      (ERESULT)((unsigned int)(res) & 0xffff)
00029 
00030 
00031 #define SEVERITY_SUCCESS  0x00
00032 #define SEVERITY_FAILED   0x80
00033 
00034 #endif // __ERESULT_H__
