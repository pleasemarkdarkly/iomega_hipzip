00008 #include <cyg/infra/diag.h>

00010 //
00011 // Basic concept of debugging:
00012 //  code is organized into debugging modules. each module has a byte associated with
00013 //  it which indicates which severity of errors are printed for that module.
00014 //  you can specify -DDEBUG_LEVEL=x on the command line, which will force all modules
00015 //  to only print errors of a certain type. -DDEBUG_LEVEL=0 will cause all debugging to
00016 //  be preprocessed out
00017 
00018 //
00019 // Debugging levels
00020 //
00021 #define DBGLEV_OFF      0x00
00022 #define DBGLEV_INFO     0x01
00023 #define DBGLEV_TRACE    0x02
00024 #define DBGLEV_WARNING  0x04
00025 #define DBGLEV_ERROR    0x08
00026 #define DBGLEV_FATAL    0x10
00027 #define DBGLEV_ASSERT   0x20
00028 
00029 #define DBGLEV_DEFAULT (DBGLEV_FATAL | DBGLEV_ERROR | DBGLEV_WARNING | DBGLEV_ASSERT)
00030 
00031 #ifndef DEBUG_LEVEL
00032 #define DEBUG_LEVEL  DBGLEV_DEFAULT
00033 #endif
00034 
00035 #if DEBUG_LEVEL==0
00036 #define DEBUG_MODULE(name)
00037 #define DEBUG_MODULE_S(name,severity)
00038 #define DEBUG_USE_MODULE(name)
00039 
00040 #define DEBUG_ENABLE(name,severity)
00041 #define DEBUG_DISABLE(name,severity)
00042 
00043 #define DBG_LINE(sep)
00044 #define DEBUG(name,severity,fmt,args...)
00045 #define DEBUGP(name,severity,fmt,args...)
00046 
00047 #define DBEN(name)
00048 #define DBEX(name)
00049 #define DBTR(name)
00050 
00051 #define DBASSERT(name,cond,fmt,args...)
00052 
00053 #else    // DEBUG_LEVEL != 0
00054 
00055 
00056 // configuration
00057 // most modules should use DEBUG_MODULE() to create their debugging
00058 // instance. DEBUG_MODULE_S overrides the default debug level
00059 #define DEBUG_MODULE(name)        char __dbg_##name = DEBUG_LEVEL
00060 #define DEBUG_MODULE_S(name,sev)  char __dbg_##name = sev
00061 #define DEBUG_USE_MODULE(name)    extern char __dbg_##name
00062 
00063 // debug level adjustment
00064 // naturally these must be used in code
00065 #define DEBUG_ENABLE(name, severity)  __dbg_##name |= (severity)
00066 #define DEBUG_DISABLE(name, severity) __dbg_##name &= ~(severity)
00067 #define DEBUG_SET(name, severities)   __dbg_##name = severities
00068 
00069 // utility macro
00070 #define DBG_LINE(suff) diag_printf(" %s:%d (in %s)%s", __FILE__,__LINE__,__FUNCTION__,suff)
00071 
00072 // general debug statement
00073 // note the ({ }) around it allows it to appear as a single expression
00074 #define DEBUG(name,severity,fmt,args...) \
00075    ({if(__dbg_##name & severity) { DBG_LINE(": "); diag_printf(fmt, ##args); }})
00076 
00077 // debug statement without header (pure)
00078 #define DEBUGP(name,severity,fmt,args...) \
00079    ({if(__dbg_##name & severity) { diag_printf(fmt, ##args); }})
00080 
00081 // trace style debug statements
00082 #define DBEN(name) if(__dbg_##name & DBGLEV_TRACE) DBG_LINE(": enter\n")
00083 #define DBEX(name) if(__dbg_##name & DBGLEV_TRACE) DBG_LINE(": exit\n")
00084 #define DBTR(name) if(__dbg_##name & DBGLEV_TRACE) DBG_LINE(": trace\n");
00085 
00086 #define DBASSERT(name,cond,fmt,args...)                 \
00087    ({if((__dbg_##name & DBGLEV_ASSERT) && !(cond)) {    \
00088        DBG_LINE(": ASSERTION FAILED\n    (" #cond);     \
00089        diag_printf(")\n" fmt "\n",##args);              \
00090        while(1);                                        \
00091     }})
00092 
00093 
00094 #endif // DEBUG_LEVEL==0
00095 
00096 #endif // __DEBUG_H__
